\section{Initialization}
\label{sec:org0d2a1a5}

This document contains a literate Emacs configuration.
The prose explains \textbf{what} and \textbf{why}, while the tangled
Emacs Lisp retains concise operational comments.
\subsection{Package bootstrap}
\label{sec:org7ef9e9c}
This section must run first. It disables implicit startup
initialization, configures package archives, and ensures
that \texttt{use-package} is available before any other configuration.
\subsubsection{Disable implicit startup initialization}
\label{sec:orgabf4cf4}
\begin{verbatim}
(setq package-enable-at-startup nil)   ;; prevent implicit init from early-init
\end{verbatim}
\subsubsection{Package system and archives}
\label{sec:org76fbd6d}
\begin{verbatim}
(require 'package)

(setq package-archives
      '(("gnu"   . "https://elpa.gnu.org/packages/")
        ("melpa" . "https://melpa.org/packages/")))
\end{verbatim}
\subsubsection{Initialization and archive refresh}
\label{sec:orgcf824e9}
\begin{verbatim}
(unless package--initialized
  (package-initialize))

(unless package-archive-contents
  (package-refresh-contents))
\end{verbatim}
\subsubsection{use-package bootstrap}
\label{sec:org196b6ab}
\texttt{use-package} is used throughout the configuration to keep
package declarations declarative and composable.

\begin{verbatim}
(unless (package-installed-p 'use-package)
  (package-install 'use-package))

(eval-when-compile
  (require 'use-package))

(setq use-package-always-ensure t)   ;; auto-install dependencies
\end{verbatim}

\subsubsection{Multiple selection indicator}
\label{sec:org35f4a61}
This adds a visible indicator when completing multiple values.

\begin{verbatim}
(defun crm-indicator (args)
  (cons (format "[CRM%s] %s"
                (replace-regexp-in-string
                 "\\`\\[.*?]\\*\\|\\[.*?]\\*\\'" ""
                 crm-separator)
                (car args))
        (cdr args)))

(advice-add #'completing-read-multiple
            :filter-args #'crm-indicator)
\end{verbatim}
\subsubsection{Minibuffer prompt protection}

\subsubsection{Orderless}
\label{sec:org4047822}
Orderless enables out-of-order pattern matching using space-
separated components.

\begin{verbatim}
(use-package orderless
  :custom
  (completion-styles '(orderless basic))
  (completion-category-defaults nil)
  (completion-category-overrides
   '((file (styles partial-completion)))))
\end{verbatim}

(use-package consult
